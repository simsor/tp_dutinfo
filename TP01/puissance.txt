fonction puissance_recur(a : réel, n : entier) retourne réel
début
    avec b : réel
    
    si n  = 0 alors retourne 1
    sinon
	si n = 1 alors retourne a
	sinon
	    b <- puissance(a, n/2)
	    si n mod 2 = 0 alors
	       retourne b * b
	    sinon
		retourne a * b * b
	    fin si
	fin si
    fin si
fin puissance_recur
