// Calcul de la somme des n premiers entiers positifs
// Entrée : n, le nombre limite > 0
// Sortie : La somme des entiers de 1 à n, ou -1 si le nombre n'est pas valide
// Pré-condition: /
// Post-Condition: /

Fonction somme_iter(n : Entier) Retourne Entier
Début
	Avec
		resultat : Entier
	
	Si n < 0 Alors Retourne -1
	resultat <- 0
	Pour i de 0 à n Faire
		resultat <- resultat + i
	Fin Faire
	
	Retourne resultat
Fin somme_iter

Fonction somme_recur(n : Entier) Retourne Entier
Début
	Si n = 0 Alors Retourne 0
	Sinon Retourne n + somme_recur(n-1)
	Fin Si
Fin somme_recur
