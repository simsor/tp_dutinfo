// Calcul de la somme des n premiers entiers positifs
// Entrée : n, le nombre limite
// Sortie : La somme des entiers de 1 à n
// Pré-condition: Nombres supérieurs à 0
// Post-Condition: /

Fonction somme_iter(n : Entier) Retourne Entier
Début
	Avec
		resultat : Entier
		
	resultat <- 0
	Pour i de 0 à n Faire
		resultat <- resultat + i
	Fin Faire
	
	Retourne resultat
Fin somme_iter

Fonction somme_recur(n : Entier) Retourne Entier
Début
	Si n = 0 Alors Retourne 0
	Sinon Retourne n + somme_recur(n-1)
	Fin Si
Fin somme_recur
